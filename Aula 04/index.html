Trecho de Código Fonte

Neste exemplo vou demonstrar um trecho de código inserido o comando document.getElementById('edtCodigo) que é um comando java script

    
        
    #include stdio.h
    #include math.h

    int main() {
        double a, b, c;
        double delta, x1, x2;

        printf("Digite os valores de a, b e c: ");
        scanf("%lf %lf %lf", &a, &b, &c);

        if (a == 0) {
            printf("Isso não é uma equação do segundo grau.\n");
            return 0;
        }

        delta = b * b - 4 * a * c;

        if (delta > 0) {
            x1 = (-b + sqrt(delta)) / (2 * a);
            x2 = (-b - sqrt(delta)) / (2 * a);
            printf("Duas raízes reais:\n");
            printf("x1 = %.2lf\n", x1);
            printf("x2 = %.2lf\n", x2);
        } else if (delta == 0) {
            x1 = -b / (2 * a);
            printf("Uma raiz real:\n");
            printf("x = %.2lf\n", x1);
        } else {
            printf("Não existem raízes reais.\n");
        }

        return 0;
    }
